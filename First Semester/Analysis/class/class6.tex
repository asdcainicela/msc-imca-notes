\section{Distancia entre conjuntos}
Sean $X, Y \subset  \Rn $ no vacíos, la distancia entre $X \text{ e } Y$ está acotado inferiormente,
\begin{equation}
	d(X,Y)= \inf \qty{\abs{x-y } \in \R : x \in X, y \in Y}
\end{equation}

\rmkb{
	Si $X \subset  M \text{ e } Y \subset M$, entonces
	$$
	d(X,Y) \geq d(M,N)
	$$
	
}
